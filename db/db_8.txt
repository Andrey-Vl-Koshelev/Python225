FROM Orders o, Salespeople s
WHERE o.snum=s.snum AND s.sname='Rifkin'

FROM Orders o
WHERE snum =
(SELECT snum
FROM Salespeople

FROM Customers

4.	Определите города проживания заказчиков с наименьшим рейтингом.
SELECT city, MIN(rating)
FROM Customers
GROUP BY city
HAVING MIN(rating) =
(SELECT MIN(rating)
FROM Customers
)

SELECT city, MAX(rating)
FROM Customers
GROUP BY city
HAVING MAX(rating) =
(SELECT MIN(rating)
FROM Customers
)
